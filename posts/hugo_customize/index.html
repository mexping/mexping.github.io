Authorを非表示にする

LAYOUTS/POSTS/single.html

33〜40をコメントアウト

            	{{/*
                {{- $author := $params.author | default .Site.Author.name | default (T "author") -}}
                {{- $authorLink := $params.authorlink | default .Site.Author.link | default .Site.Home.RelPermalink -}}
                <span class="post-author">
                    {{- $options := dict "Class" "author" "Destination" $authorLink "Title" "Author" "Rel" "author" "Icon" (dict "Class" "fas fa-user-circle fa-fw") "Content" $author -}}
                    {{- partial "plugin/link.html" $options -}}
                </span>
                */}}

LAYOUTS/_DEFAULT/summary.html

29行〜をコメントアウト

{{/*
        <span class="post-author">
            {{- $options := dict "Class" "author" "Destination" $authorLink "Title" "Author" "Rel" "author" "Icon" (dict "Class" "fas fa-user-circle fa-fw") "Content" $author -}}
            {{- partial "plugin/link.html" $options -}}
        </span>
*/}}

Tagをタイトル直下に移動

LAYOUTS/PARTIALS/footer.html

46行〜の<section>の中身をコメントアウト

{{/*
            {{- with .Params.tags -}}
                <i class="fas fa-tags fa-fw"></i>&nbsp;
                {{- range $index, $value := . -}}
                    {{- if gt $index 0 }},&nbsp;{{ end -}}
                    {{- $tag := partialCached "function/path.html" $value $value | printf "/tags/%v" | $.Site.GetPage -}}
                    <a href="{{ $tag.RelPermalink }}">{{ $tag.Title }}</a>
                {{- end -}}
            {{- end -}}
*/}}

LAYOUTS/POSTS/single.html

53行〜（カテゴリーの下）に下記を追記

                {{- $tags := slice -}}
                {{- range .Params.tags -}}
                    {{- $tag := partialCached "function/path.html" . . | printf "/tags/%v" | $.Site.GetPage -}}
                    {{- $tags = $tags | append (printf `<a href="%v"><i class="fas fa-tags fa-fw"></i>%v</a>` $tag.RelPermalink $tag.Title) -}}
                {{- end -}}
                {{- with delimit $tags "&nbsp;" -}}
                    &nbsp;<span class="post-category">
                        {{- dict "Categories" . | T "includedInCategories" | safeHTML -}}
                    </span>
                {{- end -}}

カテゴリーの前の「included in」を消す

other = "{{ .Categories }}"

LAYOUTS/POSTS/single.html

69行〜に下記を追記。更新された場合だけ、更新日を表示させる。

  {{ $lastmodstr := .Site.Params.dateformat | default "2006-01-02" | .Lastmod.Format }}

  {{ if ne $datestr $lastmodstr }}
                    {{- with .Site.Params.dateformat | default "2006-01-02" | .Lastmod.Format -}}

                {{- end -}}
  {{ end }}

  {{ $lastmodstr := .Site.Params.dateformat | default "2006-01-02" | .Lastmod.Format }}
  {{ $datestr := .Site.Params.dateformat | default "2006-01-02" | .PublishDate.Format }}

        {{- end -}}
  {{ end }}

Lazyloadの修正

LAYOUTS/PARTIALS/PLUGIN/image.html

{{/*
            data-srcset="{{ $small | safeURL }}, {{ .Src | safeURL }} 1.5x, {{ $large | safeURL }} 2x"
            data-sizes="auto"
*/}}
